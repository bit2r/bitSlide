ds_url <- "https://statkclee.github.io/yonsei/"
ds_html <- read_html(ds_url)

ds_html |> 
  html_elements('table') |> 
  html_table() %>%
  .[[1]] |> 
  slice(2:n()) |> 
  separate(X2, into = c("학습내용", "날짜"), sep = "-") |> 
  separate(학습내용, into = c("주차", "학습내용"), sep = "\\.") |> 
  select(주차, 학습내용, 날짜) |> 
  drop_na() |> 
  mutate(날짜 = str_trim(날짜),
         학습내용 = str_trim(학습내용))  |> 
  mutate(날짜 = if_else(str_detect(날짜, "^\\d"), 날짜, NA_character_))  |> 
  gt() |> 
    gt_theme_pff() |> 
    sub_missing(
      columns = 날짜,
    missing_text = "-"
    )  |> 
    cols_align("center") |> 
    opt_table_font(
      font = list(
        google_font(name = "Sunflower"),
        "NanumGothic"
      )
    ) %>% 
    tab_style(
      style = list(
        "font-variant: small-caps;"
      ),
      locations = cells_body(columns = everything())
    ) %>% 
    tab_header(
      title =
        htmltools::tagList(
          htmltools::tags$div(
            style = htmltools::css(
              `text-align` = "center"
            ),
            htmltools::HTML(
              web_image("img/curriculum_yonsei.png")
            )
          ),
          htmltools::tags$div(
            htmltools::tags$strong("연세대"),
            "데이터 사이언스 입문"
          )
        )
    ) |> 
  tab_options(
    table.width = "65%",
    heading.title.font.size = px(12)
  )

curriculum_raw <- read_excel(glue::glue("{here::here()}/data/건국대_커리큘럼.xlsx"))

curriculum <- curriculum_raw %>% 
  fill(everything(), .direction="down") %>% 
  mutate(날짜 = as.Date(날짜)) %>% 
  mutate(주차 = glue::glue("{주차} 주차"))

## 첫번째
curriculum_first <- curriculum %>% 
  filter(dplyr::row_number() <=10)

curriculum_first_gt <- curriculum_first %>%   
  gt(groupname_col = c("주차", "날짜")) %>% 
    cols_align(align = "center") %>%

    font = list(
      google_font(name = "Sunflower"),
      "NanumGothic"

    title =
